void setup() {
  Serial.begin(9600);
/*Define os pinos como saída */
  pinMode(a, OUTPUT);
  pinMode(b, OUTPUT);
  pinMode(c, OUTPUT);
  pinMode(d, OUTPUT);
  pinMode(e, OUTPUT);
  pinMode(f, OUTPUT);
  pinMode(g, OUTPUT);
  pinMode(h, OUTPUT);

Serial.println("Digite um numero: "); 
}

void loop() {
  
  if (Serial.available() > 0) {
    String input = Serial.readString();

    int soma = somarDigitos(input);

    while (soma >= 10) {
      soma = somarDigitos(String(soma));
    }

    acenderLeds(soma);
    
    delay(4000);
    
    apagarLeds(soma);
    
    Serial.println("Digite um numero: ");
  }
}

int somarDigitos(String numero) {
  int soma = 0;
  for (int i = 0; i < numero.length(); i++) {
    soma += numero[i] - '0';
  }
  return soma;
}
